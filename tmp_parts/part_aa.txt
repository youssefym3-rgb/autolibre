  const mnu=document.getElementById('userMenu');
  if(!mnu.classList.contains('hidden')){mnu.classList.add('hidden');return}
  mnu.innerHTML=`<div style="padding:10px 12px"><b>${me.name}</b><br><span style="font-size:13px;color:var(--muted)">${me.type==='pro'?'Concesionario':'Particular'} · Plan ${me.plan}</span></div><hr>
    <a onclick="go('dashboard')">▤ Mi panel</a><a onclick="go('publish')">＋ Publicar coche</a>
    <a onclick="go('favorites')">♥ Favoritos (${favIds.size})</a><a onclick="go('messages')">✉ Mensajes</a><a onclick="openMyAlerts()">🔔 Mis alertas</a>
    ${me.role==='admin'?`<hr><a onclick="go('admin')" style="color:var(--danger);font-weight:700">⚙ Administración</a>`:''}
    <hr><a onclick="logout()">⏻ Cerrar sesión</a>`;
  mnu.classList.remove('hidden');
}

function carCard(c){
  const photo=c.photos&&c.photos.length?`<img class="ph" src="${c.photos[0]}">`:`<div class="placeholder">🚗</div>`;
  const isNew=(Date.now()-c.created)<7*86400000;
  return `<div class="car-card" onclick="openDetail(${c.id})">
    <div class="car-photo">${photo}
      ${c.featured?'<span class="tag pro">Destacado</span>':''}
      ${isNew&&!c.featured?'<span class="tag new">Nuevo</span>':''}
      <button class="fav-btn ${isFav(c.id)?'on':''}" onclick="event.stopPropagation();toggleFav(${c.id})">${isFav(c.id)?'♥':'♡'}</button>
    </div>
    <div class="car-body">
      <h3 class="car-title">${c.brand} ${c.model}</h3>
      <p class="car-sub">${c.year} · ${c.province}</p>
      <div class="car-specs">
        <span class="spec">${fmt(c.km)} km</span><span class="spec">${c.fuel}</span>
        <span class="spec">${c.gear}</span><span class="spec">${c.power} CV</span>
      </div>
      <div class="car-foot">
        <div class="price">${euro(c.price)}${c.priceLabel==='good'?' <span style="background:#eafaf3;color:#0fa968;font-size:11px;font-weight:800;padding:2px 7px;border-radius:20px;vertical-align:middle">↓ Buen precio</span>':c.priceLabel==='fair'?' <span style="background:#eef4ff;color:#0a58ca;font-size:11px;font-weight:800;padding:2px 7px;border-radius:20px;vertical-align:middle">Precio justo</span>':''}</div>
        <div class="seller-mini">${c.owner&&c.owner.type==='pro'?'🏢':'👤'} ${c.owner?c.owner.name.split(' ')[0]:''}</div>
      </div>
    </div>
  </div>`;
}
